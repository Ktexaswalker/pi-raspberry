# pi-raspberry
#Ejecución de código por pulsaciones hardware

pi@raspberry:~$ sudo raspi-config     #configurar la camara YES
pi@raspberry:~$ sudo nano camera.sh   #ejecuta la camara con los parametros asignados en la linea de codigo
pi@raspberry:~$ sudo nano shutdown.py #ejecuta el programa de apagado
pi@raspberry:~$ sudo nano /etc/rc.local
pi@raspberry:~$ sudo reboot
